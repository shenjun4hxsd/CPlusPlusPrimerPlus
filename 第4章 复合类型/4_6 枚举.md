##枚举

```javascript
    enum EnemyType
    {
        Melee = 1,
        Throw,
        Magic
    };
    
    enum // 如果只打算使用常量，而不创建枚举类型的变量，则可以省略枚举类型的名称
    {
        Melee1 = 1,
        Throw1,
        Magic1
    };
    
    enum bits {one = 1, two = 2, four = 4, eight = 8};
    
    int main()
    {
        EnemyType enemyType = Magic;
        enemyType = (EnemyType)1;     // 需要进行强转
        cout << enemyType << endl;
    
        return 0;
    }
```
